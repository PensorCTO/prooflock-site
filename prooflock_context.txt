            const resultArea = document.getElementById('result-area');

        function setMode(mode) {
            // Prevent mode change if issue mode is disabled for guests
            if (mode === 'bug' && !myUuid) {
                return false;
            }
            
            currentMode = mode;
            
            // Clear all active classes
            document.querySelectorAll('.toggle-opt').forEach(el => {
                el.classList.remove('active');
            });

            if (mode === 'bug') {

                if (btnIssue) {
                    btnIssue.classList.add('active');
                    btnIssue.classList.remove('disabled');

        async function fetchTickets() {
            const listEl = document.getElementById('ticket-list');
            const statusEl = document.getElementById('history-status');
            const lookupId = myUuid || '00000000-0000-0000-0000-000000000000'; 

            const { data, error } = await supabase.rpc('get_filtered_tickets', { lookup_uuid: lookupId });

            if (error) {
                console.error(error);
                listEl.innerHTML = `<div style="color: var(--danger); text-align:center; padding-top: 150px;">Connection Error.</div>`;
                statusEl.innerText = "Offline";
                return;
            }

            statusEl.innerText = "Synced";

            if (!data || data.length === 0) {
                listEl.innerHTML = `<div style="text-align: center; color: #444; padding-top: 150px;">Log Empty. Start the conversation.</div>`;
                return;
            }

            // Render List
            listEl.innerHTML = data.map(ticket => `
                <div class="ticket-card">
                    <div class="ticket-header">
                        <span class="status-badge status-${ticket.status}">${ticket.status}</span>
                        <span class="ticket-date">${new Date(ticket.created_at).toLocaleDateString()} ‚Ä¢ ${ticket.type.toUpperCase()}</span>
                    </div>
                    <div class="ticket-msg">${escapeHtml(ticket.message)}</div>
                    
                    ${ticket.admin_reply ? `
                        <div class="admin-reply">
                            <div class="reply-avatar">PL</div>
                            <div class="reply-body">
                                <span class="reply-label">Architect Response</span>
                                <div class="reply-text">${escapeHtml(ticket.admin_reply)}</div>
                            </div>
                        </div>
                    ` : ''}
                </div>
            `).join('');
        }

        // --- 6. SUBMIT LOGIC ---
        async function submitTicket() {
            const msg = document.getElementById('message-input').value.trim();
            const btn = document.getElementById('submit-btn');

            if (!msg) return alert("Please enter a message.");
            if (currentMode === 'bug' && !myUuid) return alert("Private bugs require App ID.");

            btn.disabled = true;
            btn.innerText = "Encrypting & Sending...";

            const submissionUuid = myUuid || crypto.randomUUID();

            const { error } = await supabase.from('customer_service').insert({
                user_uuid: submissionUuid,
                type: currentMode,
                message: msg,
                status: 'open'
            });

            if (error) {
                alert("Transmission Failed: " + error.message);
                btn.disabled = false;
                btn.innerText = "Try Again";
            } else {
                document.getElementById('message-input').value = "";
                btn.innerText = "Sent ‚úÖ";
                setTimeout(() => { 
                    btn.disabled = false; 
                    setMode(currentMode); 
                    fetchTickets(); 
                }, 2000);
            }
        }

        function escapeHtml(text) {
            if (!text) return '';
            return text.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;").replace(/"/g, "&quot;").replace(/'/g, "&#039;");
        }
